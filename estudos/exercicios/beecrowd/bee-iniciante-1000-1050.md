atrap = (a + b) * c / 2
aq = b ** 2
ar = a * b
print("TRIANGULO: {:.3f}\nCIRCULO: {:.3f}\nTRAPEZIO: {:.3f}\nQUADRADO: {:.3f}\nRETANGULO: {:.3f}".format(at, ac, atrap, aq, ar))

## 1013 - O Maior 

~~~python
linha = input().split(" ")
a = int(linha[0])
b = int(linha[1])
c = int(linha[2])
maiorAB = (a + b + abs(a - b)) / 2

if maiorAB > c:
	maior = maiorAB
else:
	maior = c

maior = int(maior)
print("{} eh o maior".format(maior))
~~~
